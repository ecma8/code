1.1函数方式

	function checkName()
	{
		alert(123)
	}

    1.2变量方式

	var checkName=function()
	{
		alert(123)
	}

    1.3对象收集变量

	var checkObject={
		checkName:function(){
			alert(123)
		}
	}

    1.4对象的另外一种形式

	var checkObject=function(){};
	checkObject.checkName=function(){
		alert(123)
	}

    1.5真假对象

	var checkObject=function(){
		return{
			checkName:function(){
				alert(123)
			}
		}
	}
	var a=checkObject();
	a.checkName();

    1.6类的实现

	var a=checkObject=function(){
		this.checkName=function(){
			alert(123)
		}
	}
	var a=new checkObject();
	a.checkName();

    1.7函数的原型

	////写法一////
	var checkObject=function(){}
	checkObject.prototype.checkName=function(){
		alert(123)
	}
	var a=new checkObject();
	a.checkName();
	////写法二////
	var checkObject=function(){}
	checkObject.prototype={
		checkName:function(){
			alert(123)
		}
	}
	var a=new checkObject();
	a.checkName();

    1.8链式调用

	var checkObject=function(){}
	checkObject.prototype={
		checkName:function(){
			alert(123);
			return this;

		},
		checkEmail:function(){
			alert(456)
			return this;
		}
	}
	var a=new checkObject();
	a.checkName().checkEmail();

    1.9函数的祖先

	Function.prototype.addMethod=function(name,fn){
		this.name=fn;
	}
	var methods=function(){};
	methods.addMethod('checkName',function(){
		alert(123)
	})
	methods.checkName();

    1.10链式添加 链式调用

	Function.prototype.addMethod=function(name,fn){
		this[name]=fn;
		return this;
	}
	var methods=function(){};
	methods.addMethod('checkName',function(){
		alert(123)
		return this;
	}).addMethod('checkEmail',function(){
		alert(456)
		return this;
	})
	methods.checkName().checkEmail();

    1.11类的链式添加 链式调用

	Function.prototype.addMethod=function(name,fn){
		this.prototype[name]=fn;
		return this;
	}
	var methods=function(){};
	methods.addMethod('checkName',function(){
		alert(123)
	}).addMethod('checkEmail',function(){
		alert(456)
	})
	var a=new methods();
	a.checkName();
	a.checkEmail();
