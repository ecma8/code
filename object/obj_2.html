	var Book=function(id,bookname,price){
		this.id=id;
		this.bookname=bookname;
		this.price=price;
	}
	Book.prototype.display=function(){
		alert('这本书可以看了')
	}
	var book=new Book(10,'javascript设计模式',50);
	book.display();
	console.log(book.id);
	console.log(book.bookname);
	console.log(book.price);
